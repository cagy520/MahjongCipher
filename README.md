            //复杂加密
            string fmi=mc.EncryptAESBASE64("JKHJHDHSD54545424", "123456789");
            fmi = mc.Encrypt(fmi);
            Console.WriteLine(fmi);
            //复杂解密
            string fold = mc.Decrypt(fmi);
            fold = mc.DecryptAESBASE64(fold, "123456789");
